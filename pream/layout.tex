    leftmargin=1em,%
    rightmargin=1em,%
    middlelinewidth=1.2pt,%
    roundcorner=5pt,%
    linecolor=lightgray,%
    backgroundcolor=lightgray!20,%
    skipabove=1.6em, innertopmargin=1.6em,
    skipbelow=1em,%
    frametitlefont=\sffamily\bfseries,%
    settings={\global\stepcounter{axiom}},%
    singleextra={%
    \useasboundingbox (P) rectangle (P);
        \node[title,xshift=1cm] at (P-|O) %
            {\mdf@frametitlefont{\theaxiom}~};},%
    firstextra={%
    \useasboundingbox (P) rectangle (P);
            \node[title,xshift=1cm] at (P-|O) %
                    {~\mdf@frametitlefont{\theaxiom}~};},
}
